import streamlit as st
import pandas as pd
from data_fetcher import update_all_data, load_cached_data
from recommendation import get_recommendation
from plots import show_dashboard
import os

st.set_page_config(layout="wide", page_title="宏观驱动资产配置系统")
st.title("📊 中美宏观 - 自动持仓推荐 (20年走势图)")

st.sidebar.header("数据源管理")
# 从环境变量读取 FRED API Key（部署时设置）
default_fred_key = os.environ.get("FRED_API_KEY", "")
fred_key = st.sidebar.text_input("请输入您的 FRED API Key", type="password", value=default_fred_key)
if st.sidebar.button("更新全部数据"):
    if not fred_key:
        st.sidebar.error("请先输入 FRED API Key")
    else:
        with st.spinner("正在获取最新数据，可能需要几十秒..."):
            data = update_all_data(fred_key)
            st.sidebar.success("数据更新完成！")

data = load_cached_data()
if not data:
    st.warning("未找到缓存数据，请在左侧输入 FRED API Key 并点击「更新全部数据」")
    st.stop()

latest_us_gdp = data['us_gdp'].iloc[-1] if not data['us_gdp'].empty else 0
latest_us_cpi = data['us_cpi'].iloc[-1] if not data['us_cpi'].empty else 0
latest_us_rate = data['us_rate'].iloc[-1] if not data['us_rate'].empty else 0
latest_cn_gdp = data['cn_gdp'].iloc[-1] if not data['cn_gdp'].empty else 0
latest_cn_cpi = data['cn_cpi'].iloc[-1] if not data['cn_cpi'].empty else 0
latest_cn_rate = data['cn_rate'].iloc[-1] if not data['cn_rate'].empty else 0
latest_ppi = data['cn_ppi'].iloc[-1] if not data['cn_ppi'].empty else 0
latest_cpi = data['cn_cpi'].iloc[-1] if not data['cn_cpi'].empty else 0

rec, phase, style, spread = get_recommendation(
    latest_us_gdp, latest_us_cpi, latest_us_rate,
    latest_cn_gdp, latest_cn_cpi, latest_cn_rate,
    latest_ppi, latest_cpi
)

col1, col2 = st.columns([2, 1])
with col1:
    st.subheader("📈 20年连续宏观走势图")
    fig = show_dashboard(data['us_gdp'], data['us_cpi'], data['us_ppi'], data['us_rate'],
                         data['cn_gdp'], data['cn_cpi'], data['cn_ppi'], data['cn_rate'])
    st.plotly_chart(fig, use_container_width=True)
with col2:
    st.subheader("🔮 当前周期判断")
    st.metric("美林时钟相位", phase)
    st.metric("股票风格偏好", style)
    st.metric("中美利差(bp)", round(spread*100, 0))
    st.subheader("💼 推荐持仓")
    for product, weight in rec.items():
        st.write(f"- {product} : **{int(weight*100)}%**")
    st.caption("注：以上建议仅供参考，投资需谨慎")
